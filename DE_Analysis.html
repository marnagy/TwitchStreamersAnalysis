all_nodes = dict(G.nodes(data=True))
mature_nodes = dict()
nonmature_nodes = dict()
for node, values in all_nodes.items():
    if values['mature']:
        mature_nodes[node] = values
    else:
        nonmature_nodes[node] = values

print(f'Nonmature streamers: {len(nonmature_nodes)}')
print(f'Mature streamers: {len(mature_nodes)}')

nonmature_perc = list()
for node, values in nonmature_nodes.items():
    neighbors_mature = list()
    for neighbor_node in G[node]:
        neighbors_mature.append( all_nodes[neighbor_node]['mature'] )
    perc = sum(neighbors_mature) / len(neighbors_mature) if len(neighbors_mature) > 0 else 0
    nonmature_perc.append( perc )
print(f'Average of percentage of mature neighbors of nonmature streamers:')

import os
import pandas as pd
attrs_file_name = os.path.join('twitch_gamers', 'large_twitch_features.csv')
df_attrs = pd.read_csv(attrs_file_name)

cat_attrs_names = ['mature', 'dead_account', 'language', 'affiliate']
other_attrs = ['life_time', 'views']
attrs_dict = dict()
for col in cat_attrs_names:
    attrs_dict[col] = { val:0 for val in list(df_attrs[col].unique()) }
for col in other_attrs:
    attrs_dict[col] = list()

for node_name, values in all_nodes.items():
    for col in cat_attrs_names:
        attrs_dict[col][values[col]] += 1
    for col in other_attrs:
        attrs_dict[col].append( values[col] )

for col in cat_attrs_names:
    values = list(attrs_dict[col].keys())
    print(f'{col} -> {values}')
    plt.bar(values, [ attrs_dict[col][val] for val in values ])
    plt.grid(axis='y')
    plt.ylabel('Amount')
    plt.xlabel(col)
    plt.show()
